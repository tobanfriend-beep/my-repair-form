ระบบได้รับการตั้งค่า URL ให้พร้อมใช้งานแล้ว หากต้องการแก้ไขโค้ดที่ฝั่ง Google Drive สามารถดูได้ที่นี่:

1

โค้ด Google Apps Script ที่ใช้อยู่ปัจจุบัน

ในกรณีที่มีการแก้โค้ดที่ script.google.com นี่คือโค้ดที่คุณควรมีอยู่:

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    // ค้นหาโฟลเดอร์ปลายทาง
    var folder;
    if (data.folderId) {
      folder = DriveApp.getFolderById(data.folderId);
    } else {
      folder = DriveApp.getRootFolder();
    }
    // สร้าง/อัปโหลดภาพซ่อมบำรุง
    var repairUrl = "";
    if (data.repairImage) {
      var rBytes = Utilities.base64Decode(data.repairImage.split(",")[1]);
      var rBlob = Utilities.newBlob(rBytes, "image/jpeg", "ซ่อม_" + data.customerName + "_" + Date.now() + ".jpg");
      repairUrl = folder.createFile(rBlob).getUrl();
    }
    // สร้าง/อัปโหลดภาพหลักฐานจ่ายเงิน
    var paymentUrl = "";
    if (data.paymentImage) {
      var pBytes = Utilities.base64Decode(data.paymentImage.split(",")[1]);
      var pBlob = Utilities.newBlob(pBytes, "image/jpeg", "จ่าย_" + data.customerName + "_" + Date.now() + ".jpg");
      paymentUrl = folder.createFile(pBlob).getUrl();
    }
    // ค้นหาหรือสร้าง Google Sheet "บันทึกซ่อมบำรุง" สำหรับเก็บตารางข้อมูล
    var sheetName = "บันทึกข้อมูลรถซ่อมบำรุง";
    var files = folder.getFilesByName(sheetName);
    var ss;
    if (files.hasNext()) {
      ss = SpreadsheetApp.open(files.next());
    } else {
      ss = SpreadsheetApp.create(sheetName);
      var file = DriveApp.getFileById(ss.getId());
      folder.addFile(file);
      DriveApp.getRootFolder().removeFile(file);
    }
    var sheet = ss.getSheets()[0];
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(["วันเวลาซ่อม", "ชื่อลูกค้า", "เบอร์ติดต่อ", "LINE ID", "ประเภทบริการ", "ข้อมูลรถ", "รายละเอียดอาการ", "ราคา (บาท)", "การชำระเงิน", "รูปงานซ่อม", "รูปภาพสลิป"]);
    }
    sheet.appendRow([
      data.dateTime,
      data.customerName,
      data.customerPhone,
      data.lineId,
      data.serviceType,
      data.carInfo,
      data.serviceDetails,
      data.price,
      data.paymentType,
      repairUrl,
      paymentUrl
    ]);
    return ContentService.createTextOutput(JSON.stringify({status: "success", message: "อัปโหลดเรียบร้อยแล้ว!", repairUrl: repairUrl, paymentUrl: paymentUrl})).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
